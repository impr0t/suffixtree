132 std::vector<Suffix*> SuffixArray::getAllSubstringPositions(std::string text)
133 {
134  long l = 0;
135  long r = this->last.size();
136 
137  // use a binary search to discover our starting position to scan from.
138  while (l < r) {
139  long mid = (l + r) / 2;
140  if (text.compare(this->last[mid]->getSuffixText()) > 0) {
141  l = mid + 1;
142  } else {
143  r = mid;
144  }
145  }
146 
147  // scan forward and add records until we don't find our substring anymore.
148  long startInterval = l;
149  std::vector<Suffix*> result;
150  for (long i = startInterval; i < this->last.size(); i++) {
151  if (this->last[i]->getSuffixText().find(text) == 0) {
152  result.push_back(this->last[i]);
153  } else {
154  break;
155  }
156  }
157 
158  return result;
