130 SuffixTreeNode *SuffixTree::findSubstring(SuffixTreeNode *cur, std::string text, int c)
131 {
132  // if node is null, well... not much we can do.
133  if (cur != NULL) {
134 
135  // go through all the children of the current node.
136  for (auto elem : cur->children) {
137  // if the first character tag matches investigate.
138  if (elem.first == text[c]) {
139  // pull the suffix from the tree.
140  std::string suffix = elem.second->edgeText;
141  // if they are exact matches, or the search text
142  // exists INSIDE of the suffix, return the suffix.
143  if ((suffix.compare(text) == 0)
144  || (charCheck(suffix, text.substr(c)) == 0))
145  {
146  return cur;
147  }
148  else
149  {
150  return findSubstring(elem.second, text, ++c);
151  }
152  }
153  }
154  }
155  return NULL;
156 }

167 int SuffixTree::charCheck(std::string t1, std::string t2)
168 {
169  // if the search string is bigger than the suffix no subtring
170  // can be found.
171  if (t1.length() < t2.length()) return -1;
172
